111 static inline void filter_data() {
112  tps1_value = (tps1_value + filter_buffer(buf[obufn] + TPS1_ADC_OFFSET, ADC_BUFFER_SIZE, ADC_CHANNELS)) / 2;
113  tps2_value = (tps2_value + filter_buffer(buf[obufn] + TPS2_ADC_OFFSET, ADC_BUFFER_SIZE, ADC_CHANNELS)) / 2;
114  brake_value = (brake_value + filter_buffer(buf[obufn] + BRAKE_ADC_OFFSET, ADC_BUFFER_SIZE, ADC_CHANNELS)) / 2;
115  SC_value = (SC_value + filter_buffer(buf[obufn] + SC_ADC_OFFSET, ADC_BUFFER_SIZE, ADC_CHANNELS)) / 2;
116 
117  if (calibrate) {
118  if (tps1_value < tps1_low) tps1_low = tps1_value;
119  if (tps1_value > tps1_max) tps1_max = tps1_value;
120  if (tps2_value < tps2_low) tps2_low = tps2_value;
121  if (tps2_value > tps2_max) tps2_max = tps2_value;
122  if (brake_value < brake_low) brake_low = brake_value;
123  if (brake_value > brake_max) brake_max = brake_value;
124  }
125 
126  tps1_adc_percentage = map(tps1_value, tps1_low, tps1_max, 0, 100);
127  tps2_adc_percentage = map(tps2_value, tps2_low, tps2_max, 0, 100);
128  brake_adc_percentage = map(brake_value, brake_low, brake_max, 0, 100);
129 
130  // check APPS plausibility
131  if (abs(tps1_adc_percentage - tps2_adc_percentage) > APPS_PLAUS_RANGE)
132  apps_adc_plausibility = false;
133  else
134  apps_adc_plausibility = true;
135 
136  // check APPS + brake plausibility (BSPD)

138  brake_adc_plausibility = false;
139  else if (!brake_adc_percentage)
140  brake_adc_plausibility = true;

143 void ADC_Handler() {
144  int f = ADC->ADC_ISR;
145  if (f & (1 << 27)){
